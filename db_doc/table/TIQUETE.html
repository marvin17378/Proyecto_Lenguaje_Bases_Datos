TRIGGER TRG_PREVENIR_SOBREVENTA
BEFORE INSERT ON TIQUETE
FOR EACH ROW
DECLARE
    v_asientos_ocupados NUMBER;
BEGIN
    SELECT COUNT(*) INTO v_asientos_ocupados
    FROM TIQUETE
    WHERE id_funcion = :NEW.id_funcion AND id_asiento = :NEW.id_asiento;

    IF v_asientos_ocupados > 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'Error: El asiento seleccionado ya está vendido para esta función.');
    END IF;
END;
